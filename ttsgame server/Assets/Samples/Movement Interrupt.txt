private bool TestInterrupt(InterruptData _data)
{
    if (!_data.SearchTypesFor(InterruptEventType.Movement_Start, InterruptEventType.Movement_Passby))
       return false;

    if(!Utilities.TargetWithinRange(new Vector2(0,1), GameManager.Entities[_data.TriggererID].MapPosition, 7))
    {
        int pathCost = (int)_data.ExtraInterruptData[0];
        IList pathdata = (IList)_data.ExtraInterruptData[1];

        for (int i = 0; i < pathdata.Count; i++)
            if (Utilities.TargetWithinRange(new Vector2(0, 1), (pathdata[i] as Node).Position, 7))
            {
                Debug.Log("Interrupted Along Path");
                return true;
            }
        return false;
    }
    Debug.Log("Interrupted");
    return true;
}
